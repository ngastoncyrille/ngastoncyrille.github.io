Avant-propos

La télédétection représente un domaine captivant où les données satellitaires jouent un rôle crucial en fournissant des informations essentielles sur notre environnement et nos ressources naturelles. Ce document explore la classification non supervisée d’images satellites, une méthode indispensable pour extraire des informations pertinentes à partir de vastes ensembles de données.

La classification non supervisée consiste à regrouper des pixels similaires d’une image en groupes homogènes basés principalement sur leurs caractéristiques spectrales, sans données de référence préalables. Elle est couramment utilisée pour :

  • La surveillance environnementale
  • La cartographie de la couverture terrestre
  • La gestion des catastrophes naturelles

Zone d’étude : Commune de Bama, Burkina Faso (ouest du pays).
Données : Image Landsat 8, acquisition du 25 janvier 2024.

2.2 Normalisation et suppression des valeurs aberrantes

Code
# Quantiles pour l'étirement linéaire
q_low  <- 0.02
q_high <- 0.98

# Appliquer par bande
for (b in names(landsat_mask)) {
  vals      <- values(landsat_mask[[b]])
  v_low     <- quantile(vals, q_low,  na.rm = TRUE)
  v_high    <- quantile(vals, q_high, na.rm = TRUE)
  vals_norm <- (vals - v_low) / (v_high - v_low)
  vals_norm <- pmax(0, pmin(1, vals_norm))  # Clamp [0, 1]
  values(landsat_mask[[b]]) <- vals_norm
}

2.3 Préparation de la matrice pour le clustering

Code
# Convertir le raster en matrice
mat <- as.matrix(values(landsat_mask))

# Supprimer les NA
valid_idx <- complete.cases(mat)
mat_clean <- mat[valid_idx, ]

# Échantillonnage (pour accélérer le calcul)
set.seed(42)
n_sample <- 15000
idx_sample <- sample(nrow(mat_clean), min(n_sample, nrow(mat_clean)))
mat_sample <- mat_clean[idx_sample, ]

cat("Pixels valides :", nrow(mat_clean), "\n")
cat("Taille échantillon :", nrow(mat_sample), "\n")

3. Classification K-means

3.1 Choix du nombre optimal de clusters

Code
# Méthode de l'indice de silhouette
k_range <- 2:8
sil_scores <- numeric(length(k_range))

for (i in seq_along(k_range)) {
  k      <- k_range[i]
  km     <- kmeans(mat_sample, centers = k, nstart = 10, iter.max = 100)
  sil    <- silhouette(km$cluster, dist(mat_sample[sample(nrow(mat_sample), 500), ]))
  sil_scores[i] <- mean(sil[, 3])
  cat("k =", k, "| Silhouette moyen =", round(sil_scores[i], 4), "\n")
}

# Visualisation des scores
sil_df <- data.frame(k = k_range, silhouette = sil_scores)

ggplot(sil_df, aes(x = k, y = silhouette)) +
  geom_line(color = "#00b4d8", linewidth = 1.2) +
  geom_point(color = "#48cae4", size = 4) +
  geom_point(data = sil_df[which.max(sil_df$silhouette), ],
             color = "#ffd166", size = 6, shape = 18) +
  labs(
    title    = "Choix du nombre optimal de clusters",
    subtitle = "Méthode de l'indice de silhouette moyen",
    x        = "Nombre de clusters (k)",
    y        = "Indice de silhouette moyen"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.background  = element_rect(fill = "#0d1b2a", color = NA),
    panel.background = element_rect(fill = "#112240", color = NA),
    panel.grid = element_line(color = "#1a3a6b"),
    axis.text  = element_text(color = "#cdd6f4"),
    axis.title = element_text(color = "#90a8c0"),
    plot.title    = element_text(color = "#90e0ef", face = "bold"),
    plot.subtitle = element_text(color = "#7090a0")
  )

3.2 Application du K-means

Code
# Paramètres optimaux (adapter selon le résultat ci-dessus)
k_optimal <- 5

set.seed(42)
km_result <- kmeans(
  x       = mat_clean,
  centers = k_optimal,
  nstart  = 25,
  iter.max = 200
)

# Résultats
cat("K-means avec k =", k_optimal, "\n")
cat("Inertie intra-cluster :", round(km_result$tot.withinss, 2), "\n")
cat("Répartition des clusters :\n")
print(table(km_result$cluster))

# Créer le raster de classification
class_rast <- rast(landsat_mask[[1]])  # Copier la structure
