```{r radar-competences}
library(plotly)

categories <- c(
  "Hydrologie &<br>Modélisation",
  "SIG &<br>Télédétection",
  "Analyse R &<br>Python",
  "Machine<br>Learning",
  "Gestion<br>des Risques",
  "GIRE &<br>Planification",
  "Visualisation<br>des données",
  "Formation &<br>Communication"
)

niveaux <- c(90, 85, 88, 75, 82, 80, 78, 85)
# Fermer le polygone
categories_plot <- c(categories, categories[1])
niveaux_plot    <- c(niveaux, niveaux[1])

fig <- plot_ly(
  type   = 'scatterpolar',
  fill   = 'toself',
  mode   = 'lines+markers'
) |>
  add_trace(
    r     = niveaux_plot,
    theta = categories_plot,
    name  = "Niveau",
    line  = list(color = '#00b4d8', width = 2.5),
    marker = list(color = '#48cae4', size = 8),
    fillcolor = 'rgba(0, 180, 216, 0.18)'
  ) |>
  layout(
    paper_bgcolor = 'rgba(0,0,0,0)',
    plot_bgcolor  = 'rgba(0,0,0,0)',
    polar = list(
      bgcolor = 'rgba(17,34,64,0.6)',
      radialaxis = list(
        visible = TRUE, range = c(0, 100),
        gridcolor = '#1a3a6b', color = '#7090a0',
        tickfont = list(color = '#7090a0', size = 11)
      ),
      angularaxis = list(
        color = '#90a8c0',
        tickfont = list(color = '#cdd6f4', size = 12)
      )
    ),
    legend = list(font = list(color = '#cdd6f4')),
    margin = list(l = 80, r = 80, t = 40, b = 40)
  )

fig
```

## 📦 Packages R principaux

```{r packages-r}
library(ggplot2)
library(dplyr)

pkgs <- data.frame(
  Package  = c("terra","sf","stars","leaflet","ggplot2","plotly",
               "airGR","tidyr","dplyr","cluster","randomForest",
               "rmarkdown","lubridate","ncdf4","raster"),
  Domaine  = c("SIG","SIG","SIG","SIG","Visualisation","Visualisation",
               "Hydrologie","Données","Données","ML","ML",
               "Publication","Données","Climatologie","SIG"),
  Niveau   = c(85,80,70,82,88,80,85,90,90,75,72,85,88,75,70)
)

couleurs_domaine <- c(
  "SIG"          = "#f4a261",
  "Visualisation"= "#48cae4",
  "Hydrologie"   = "#0096c7",
  "Données"      = "#06d6a0",
  "ML"           = "#c4b5fd",
  "Publication"  = "#90e0ef",
  "Climatologie" = "#ffd166"
)

pkgs <- pkgs |> arrange(desc(Niveau))

ggplot(pkgs, aes(x = reorder(Package, Niveau), y = Niveau, fill = Domaine)) +
  geom_col(width = 0.7) +
  geom_text(aes(label = paste0(Niveau, "%")), 
            hjust = -0.2, size = 3, color = "#cdd6f4") +
  coord_flip() +
  scale_fill_manual(values = couleurs_domaine) +
  scale_y_continuous(limits = c(0, 105), breaks = seq(0, 100, 25)) +
  labs(
    title    = "Packages R — Niveau de maîtrise",
    subtitle = "Évaluation personnelle par domaine d'utilisation",
    x        = NULL, y = "Niveau (%)",
    fill     = "Domaine"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    plot.background  = element_rect(fill = "#0d1b2a", color = NA),
    panel.background = element_rect(fill = "#112240", color = NA),
    panel.grid.major.x = element_line(color = "#1a3a6b", linewidth = 0.4),
    panel.grid.major.y = element_blank(),
    panel.grid.minor   = element_blank(),
    axis.text  = element_text(color = "#cdd6f4"),
    axis.title = element_text(color = "#90a8c0"),
    plot.title    = element_text(color = "#90e0ef", face = "bold"),
    plot.subtitle = element_text(color = "#7090a0"),
    legend.background = element_rect(fill = "#112240", color = NA),
    legend.text  = element_text(color = "#cdd6f4"),
    legend.title = element_text(color = "#90a8c0")
  )
```
